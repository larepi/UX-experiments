Initialize:

import { CloudBees } from '@cloudbees/feature-management';

const client = new CloudBees({
  sdkKey: 'sdk-dev-f3a9c2b1-••••••••',
  environment: 'development'
});

const on = await client.isEnabled('new-checkout-flow');
if (on) { renderNewCheckout(); }
else     { renderLegacyCheckout(); }
npm install @cloudbees/feature-management-react
import { CloudBeesProvider, useFlag }
  from '@cloudbees/feature-management-react';

function App() {
  return (
    <CloudBeesProvider sdkKey="sdk-dev-••••••••">
      <MyApp />
    </CloudBeesProvider>
  );
}
function Checkout() {
  const on = useFlag('new-checkout-flow');
  return on ? <NewCheckout /> : <LegacyCheckout />;
}
